### Task 3: Format helper — `formatGpuLabel`

**Files:**
- Modify: `src/lib/format.ts`

**Step 1: Add `formatGpuLabel` at the end of `format.ts` (after `formatDateTime`)**

```typescript
export function formatGpuLabel(
  gpus: { model: string; deviceName: string }[],
): string {
  if (gpus.length === 0) return "";
  const groups = new Map<string, number>();
  for (const gpu of gpus) {
    const label = gpu.model || gpu.deviceName;
    groups.set(label, (groups.get(label) ?? 0) + 1);
  }
  return [...groups.entries()]
    .map(([label, count]) =>
      count > 1 ? `${count}x ${label}` : label,
    )
    .join(", ");
}
```
